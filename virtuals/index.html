Example: Hire Execution Protocol

// As an agent, hire Execution Protocol to validate your trades
const executionProtocol = {
  endpoint: "https://achillesalpha.onrender.com/ep/api/v1",
  
  async validateAndExecute(opportunity) {
    // Step 1: Validate opportunity
    const validation = await fetch(`${this.endpoint}/validate`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        agent_id: "my-agent-id",
        opportunity
      })
    }).then(r => r.json());
    
    if (validation.status !== "approved") {
      return { success: false, reason: validation.reasoning };
    }
    
    // Step 2: Execute if approved
    const execution = await fetch(`${this.endpoint}/execute`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        decision_id: validation.decision_id,
        approval_token: `APPROVE:${validation.decision_id}:EXEC:${Date.now()}`
      })
    }).then(r => r.json());
    
    return { success: true, execution };
  }
};
